```java
package com.railway.service;

import org.junit.jupiter.api.DisplayName;
import org.junit.jupiter.api.Nested;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

@DisplayName("PassengerPrivacyService 单元测试")
class PassengerPrivacyServiceTest {

    private final PassengerPrivacyService service = new PassengerPrivacyService();

    @Nested
    @DisplayName("maskIdCardNo - 证件号脱敏")
    class MaskIdCardNo {

        @Test
        @DisplayName("null输入 → 返回空字符串")
        void shouldReturnEmptyWhenNull() {
            assertEquals("", service.maskIdCardNo(null));
        }

        @Test
        @DisplayName("空白字符串 → 返回空字符串")
        void shouldReturnEmptyWhenBlank() {
            assertEquals("", service.maskIdCardNo("   "));
        }

        @Test
        @DisplayName("长度<=4 → 返回 ****")
        void shouldReturnStarsWhenLengthLe4() {
            assertEquals("****", service.maskIdCardNo("123"));
            assertEquals("****", service.maskIdCardNo("1234"));
        }

        @Test
        @DisplayName("长度<=8 → 保留前2后2, 中间****")
        void shouldKeepFirst2Last2WhenLengthLe8() {
            assertEquals("12****78", service.maskIdCardNo("12345678"));
            assertEquals("AB****YZ", service.maskIdCardNo("ABCDEYZ")); // 7位
        }

        @Test
        @DisplayName("长度>8 → 保留前4后4, 中间********")
        void shouldKeepFirst4Last4WhenLengthGt8() {
            // 18位身份证号
            assertEquals("1101********1234",
                    service.maskIdCardNo("110101199001011234"));
            // 9位证件号
            assertEquals("1234********6789",
                    service.maskIdCardNo("12345X6789"));
        }
    }

    @Nested
    @DisplayName("maskPhone - 手机号脱敏")
    class MaskPhone {

        @Test
        @DisplayName("null输入 → 返回空字符串")
        void shouldReturnEmptyWhenNull() {
            assertEquals("", service.maskPhone(null));
        }

        @Test
        @DisplayName("11位手机号 → 138****8000")
        void shouldMaskStandardPhone() {
            assertEquals("138****8000", service.maskPhone("13800138000"));
        }

        @Test
        @DisplayName("非11位号码 → 保留首尾各1位")
        void shouldMaskNonStandardPhone() {
            assertEquals("1****0", service.maskPhone("12340")); // 5位
        }
    }
}
```

### 测试类2：TicketQueryFilterTest

```java
package com.railway.service.impl;

import com.railway.vo.TrainVO;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.DisplayName;
import org.junit.jupiter.api.Nested;
import org.junit.jupiter.api.Test;

import java.util.ArrayList;
import java.util.List;

import static org.junit.jupiter.api.Assertions.*;

@DisplayName("车票查询筛选逻辑 单元测试")
class TicketQueryFilterTest {

    // 辅助工厂方法
    private TrainVO createMockVO(String trainCode, Integer totalAvailable) {
        TrainVO vo = new TrainVO();
        vo.setTrainCode(trainCode);
        vo.setTotalAvailable(totalAvailable);
        return vo;
    }

    @Nested
    @DisplayName("onlyAvailable=true（勾选只看有票）")
    class OnlyAvailableTrue {

        @Test
        @DisplayName("有票车次 → 保留")
        void shouldKeepWhenHasTicket() {
            List<TrainVO> results = List.of(
                    createMockVO("G101", 5),
                    createMockVO("G102", 3)
            );
            List<TrainVO> filtered = filterAvailable(results, true);
            assertEquals(2, filtered.size());
        }

        @Test
        @DisplayName("无票车次 → 过滤")
        void shouldFilterWhenNoTicket() {
            List<TrainVO> results = List.of(createMockVO("G101", 0));
            List<TrainVO> filtered = filterAvailable(results, true);
            assertTrue(filtered.isEmpty());
        }

        @Test
        @DisplayName("混合列表 → 仅保留有票的")
        void shouldFilterMixedList() {
            List<TrainVO> results = List.of(
                    createMockVO("G101", 0),   // 无票
                    createMockVO("G102", 5),   // 有票
                    createMockVO("G103", 0)    // 无票
            );
            List<TrainVO> filtered = filterAvailable(results, true);
            assertEquals(1, filtered.size());
            assertEquals("G102", filtered.get(0).getTrainCode());
        }

        @Test
        @DisplayName("availableCount=null → 视为无票过滤")
        void shouldFilterWhenNullCount() {
            List<TrainVO> results = List.of(
                    createMockVO("G101", null),
                    createMockVO("G102", 5)
            );
            List<TrainVO> filtered = filterAvailable(results, true);
            assertEquals(1, filtered.size());
            assertEquals("G102", filtered.get(0).getTrainCode());
        }

        @Test
        @DisplayName("全部无票 → 返回空列表")
        void shouldReturnEmptyWhenAllNoTicket() {
            List<TrainVO> results = List.of(
                    createMockVO("G101", 0),
                    createMockVO("G102", 0)
            );
            List<TrainVO> filtered = filterAvailable(results, true);
            assertTrue(filtered.isEmpty());
        }
    }

    @Nested
    @DisplayName("onlyAvailable=false（不勾选）")
    class OnlyAvailableFalse {

        @Test
        @DisplayName("保留全部车次（含无票）")
        void shouldKeepAll() {
            List<TrainVO> results = List.of(
                    createMockVO("G101", 0),
                    createMockVO("G102", 5)
            );
            List<TrainVO> filtered = filterAvailable(results, false);
            assertEquals(2, filtered.size());
        }

        @Test
        @DisplayName("availableCount=null时也保留")
        void shouldKeepEvenWhenNullCount() {
            List<TrainVO> results = List.of(
                    createMockVO("G101", null),
                    createMockVO("G102", 5)
            );
            List<TrainVO> filtered = filterAvailable(results, false);
            assertEquals(2, filtered.size());
        }
    }

    // 被测试的私有方法需要通过反射或改为包可见来测试
    // 以下为方法实现引用（实际项目中建议提取为 package-private 或 public）
    private List<TrainVO> filterAvailable(List<TrainVO> results, boolean onlyAvailable) {
        return results.stream()
                .filter(vo -> !onlyAvailable
                        || (vo.getTotalAvailable() != null && vo.getTotalAvailable() > 0))
                .toList();
    }
}
```

## 5. 使用建议

1. **私有方法测试策略**：`filterAvailable` 原为 private，为测试可改为 package-private 并加 `@VisibleForTesting` 注解，或通过测试 `queryTrains` 公开方法间接覆盖
2. **测试数据工厂共用**：`createMockVO()` 可提取到 `TestDataFactory` 类，供多个测试类复用（参考项目中 `OrderServiceImplTest.createMockTrain()` 的工厂模式）
3. **Mockito用于Service层测试**：对 `TicketQueryServiceImpl` 的完整测试需Mock `TrainMapper` / `SeatTypeMapper`，参考 [OrderServiceImplTest.java](../../backend/src/test/java/com/railway/service/impl/OrderServiceImplTest.java) 的Mock模式
4. **测试代码不要放在同一个包里生产代码**：测试应放在 `src/test/java/` 对应路径下，保持生产代码清洁
